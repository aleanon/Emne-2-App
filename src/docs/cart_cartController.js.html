/**
 * Legger et produkt til i handlekurven eller oppdaterer mengden hvis produktet allerede finnes.
 *
 * Funksjonen søker etter et eksisterende produkt i handlekurven basert på produktnavnet. Hvis
 * produktet allerede er i handlekurven, økes mengden med den angitte verdien. Hvis ikke,
 * legges produktet til som et nytt element. Etterpå oppdateres handlekurven.
 *
 * @function
 * @name addToCart
 * @param {string} name - Navnet på produktet som skal legges til i handlekurven.
 * @param {number} price - Prisen på produktet.
 * @param {number} productId - ID-en til produktet.
 * @param {number} [quantity=1] - Mengden av produktet som skal legges til. Standard er 1.
 * @example
 * // Legg til et produkt med navnet "Sjokoladekake" og en mengde på 2:
 * addToCart("Sjokoladekake", 150, 101, 2);
 */
function addToCart(name, price, productId, quantity = 1) {
  const cart = model.inputs.shoppingCart.products;
  const existingItem = cart.find((item) => item.name === name);
  if (existingItem) {
    existingItem.quantity += quantity;
  } else {
    cart.push({ name, price, quantity, productId });
  }
  updateCart('add');
}

/**
 * Fjerner et produkt fra handlekurven basert på produktnavnet.
 *
 * Funksjonen filtrerer ut produktet med det angitte navnet fra handlekurven og oppdaterer
 * handlekurven etter at produktet er fjernet.
 *
 * @function
 * @name removeFromCart
 * @param {string} name - Navnet på produktet som skal fjernes fra handlekurven.
 * @example
 * // Fjern produktet "Sjokoladekake" fra handlekurven:
 * removeFromCart("Sjokoladekake");
 */
function removeFromCart(name) {
  const cart = model.inputs.shoppingCart.products;
  model.inputs.shoppingCart.products = cart.filter(
    (item) => item.name !== name,
  );
  updateCart('remove');
}

/**
 * Tømmer handlekurven ved å fjerne alle produkter.
 *
 * Denne funksjonen setter handlekurvens produktliste til en tom liste, som effektivt
 * fjerner alle elementer i handlekurven. Deretter oppdateres handlekurven for å reflektere
 * at den er tom.
 *
 * @function
 * @name clearCart
 * @example
 * // Tøm handlekurven:
 * clearCart();
 */
function clearCart() {
  model.inputs.shoppingCart.products = [];
  // updateView();
  updateCart('clear');
}

/**
 * Oppdaterer handlekurvens visning og viser en melding basert på handlingen.
 *
 * Denne funksjonen oppdaterer antall produkter i handlekurven, viser en
 * varselmelding basert på hvilken handling som ble utført (`add`, `remove` eller `clear`),
 * og gir en animasjonseffekt på handlekurvlenken. Hvis brukeren er på handlekurvsiden,
 * renderes handlekurven på nytt for å vise oppdatert innhold.
 *
 * @function
 * @name updateCart
 * @param {string} action - Handlingen som utføres på handlekurven. Kan være `add`, `remove` eller `clear`.
 * @example
 * // Oppdater handlekurven etter å ha lagt til et produkt:
 * updateCart('add');
 */
function updateCart(action) {
  const cartLink = document.getElementById('cartLink');
  const cartCount = document.getElementById('cartCount');
  const count = model.inputs.shoppingCart.products.reduce(
    (sum, item) => sum + item.quantity,
    0,
  );

  if (cartCount) {
    cartCount.textContent = count;
  }

  if (action === 'add') {
    showCartNotification('Produkt lagt til i handlekurven!');
  } else if (action === 'remove') {
    showCartNotification('Produkt fjernet fra handlekurven!');
  } else if (action === 'clear') {
    showCartNotification('Handlekurven er nå tom!');
  }

  if (cartLink) {
    cartLink.classList.add('bounce', 'pulse');
    setTimeout(() => {
      cartLink.classList.remove('bounce', 'pulse');
    }, 600);
  }

  if (model.app.currentPageIndex === shoppingCart) {
    renderCart();
  }
}

/**
 * Viser handlekurvsiden ved å oppdatere visningen til handlekurvens oversikt.
 *
 * Denne funksjonen setter `currentPageIndex` til `shoppingCart`, endrer
 * handlekurvens visningsmodus (`case`) til 'Overview', og kaller `updateView()`
 * for å vise handlekurvens innhold.
 *
 * @function
 * @name showShoppingCart
 * @example
 * // Bytt til handlekurvens oversiktsside:
 * showShoppingCart();
 */
function showShoppingCart() {
  model.app.currentPageIndex = shoppingCart;
  model.inputs.shoppingCart.case = 'Overview';
  updateView();
}

/**
 * Henter alle produkter i handlekurven.
 *
 * Denne funksjonen returnerer en liste over alle produkter som for øyeblikket er i handlekurven.
 *
 * @function
 * @name getCartItems
 * @returns {Array<Object>} En liste over produktobjekter i handlekurven.
 * @example
 * // Hent alle produkter fra handlekurven:
 * const items = getCartItems();
 */
function getCartItems() {
  return model.inputs.shoppingCart.products;
}

// Gjør funksjonene globalt tilgjengelige
window.addToCart = addToCart;
window.removeFromCart = removeFromCart;
window.clearCart = clearCart;
window.showShoppingCart = showShoppingCart;
window.getCartItems = getCartItems;
